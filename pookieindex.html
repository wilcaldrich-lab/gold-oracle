    <script>

        const PROXY = 'https://api.allorigins.win/get?url=';

        const COMMODITIES = [

            { id: 'btc', label: 'Bitcoin (Scarcity Benchmark)', usd: 0, isBtc: true },

            { id: 'gold', label: 'Gold (Troy Ounce)', usd: 2714 },

            { id: 'silver', label: 'Silver (Troy Ounce)', usd: 31 },

            { id: 'oil', label: 'Crude Oil (WTI Barrel)', usd: 72 },

            { id: 'f35', label: 'F-35 Lightning II', usd: 82500000 },

            { id: 'carrier', label: 'Ford-Class Carrier', usd: 13000000000 }

        ];


        let btcPrice = 96000;


        async function fetchPrices() {

            try {

                const res = await fetch('https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=usd');

                const data = await res.json();

                btcPrice = data.bitcoin.usd;

                document.getElementById('connectionStatus').innerHTML = '<span class="status-dot status-online"></span>NET_STATUS: OPERATIONAL';

                renderPrices();

            } catch (e) {

                document.getElementById('connectionStatus').textContent = 'NET_STATUS: OFFLINE_MODE';

                renderPrices();

            }

        }


        function renderPrices() {

            const grid = document.getElementById('priceGrid');

            grid.innerHTML = COMMODITIES.map(c => {

                const currentUsd = c.isBtc ? btcPrice : c.usd;

                const btcVal = (currentUsd / btcPrice).toFixed(8);

                return `

                    <div class="price-chamber">

                        <div class="price-label">${c.label}</div>

                        <div class="price-value">$${currentUsd.toLocaleString()}</div>

                        <div class="btc-conversion">₿ ${btcVal}</div>

                    </div>`;

            }).join('');

        }


        async function fetchNews() {

            const newsGrid = document.getElementById('newsGrid');

            const feeds = [

                { name: 'ZeroHedge', url: 'https://cms.zerohedge.com/fullrss2.xml', cat: 'GEOPOL' },

                { name: 'GoldSeek', url: 'https://news.goldseek.com/rss.php', cat: 'METALS' }

            ];


            let articles = [];

            for (const feed of feeds) {

                try {

                    const res = await fetch(`${PROXY}${encodeURIComponent(feed.url)}`);

                    const data = await res.json();

                    const parser = new DOMParser();

                    const xml = parser.parseFromString(data.contents, "text/xml");

                    const items = xml.querySelectorAll("item");

                    items.forEach((item, idx) => {

                        if (idx < 3) articles.push({

                            title: item.querySelector("title").textContent,

                            link: item.querySelector("link").textContent,

                            source: feed.name,

                            tag: feed.cat

                        });

                    });

                } catch (e) { console.error("Feed error", e); }

            }

            newsGrid.innerHTML = articles.map(a => `

                <div class="news-card">

                    <span class="tag">${a.tag}</span> <strong>${a.source}</strong><br>

                    <a href="${a.link}" target="_blank" style="color:white; text-decoration:none;">${a.title}</a>

                </div>`).join('');

        }


        function updateClock() {

            document.getElementById('timestamp').textContent = `SYS_TIME: ${new Date().toUTCString()}`;

        }


        fetchPrices();

        fetchNews();

        setInterval(updateClock, 1000);
